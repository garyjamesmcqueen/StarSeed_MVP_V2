'use client'


import { useGLTF } from '@react-three/drei'

import { useMemo, useRef, useEffect } from 'react'

import { useFrame } from '@react-three/fiber'

import * as THREE from 'three'


type Contribution = {

  id: string

  createdAt: number

  alias: string

  emoji: string

  message: string

  quantity: number

}


export default function Model({

  modelPath,

  contributions,

  skullPulseTrigger,

}: {

  modelPath: string

  contributions: Contribution[]

  skullPulseTrigger: number

}) {

  const { scene } = useGLTF(modelPath)


  const spinRef = useRef<THREE.Group>(null!)

  const baseRef = useRef<THREE.InstancedMesh>(null!)

  const glowRef = useRef<THREE.InstancedMesh>(null!)


  const baseMaterialRef = useRef<THREE.MeshPhysicalMaterial>(null!)

  const glowMaterialRef = useRef<THREE.MeshBasicMaterial>(null!)


  const displayCountRef = useRef(0)

  const contributionPulseRef = useRef<number[]>([])

  const phaseOffsetsRef = useRef<number[]>([])

  const speedOffsetsRef = useRef<number[]>([])


  const lastPulseTriggerRef = useRef(skullPulseTrigger)

  const structurePulseRef = useRef(0)


  const baseColor = useMemo(() => new THREE.Color('#1e3a8a'), [])

  const shimmerColor = useMemo(() => new THREE.Color('#7dd3fc'), [])

  const contributionColor = useMemo(() => new THREE.Color('#ffffff'), [])

  const farColor = useMemo(() => new THREE.Color('#0f1f4d'), [])


  const tempObject = useMemo(() => new THREE.Object3D(), [])

  const tempColor = useMemo(() => new THREE.Color(), [])


  // --------------------------------------------------

  // POINT SAMPLING

  // --------------------------------------------------


  const rawPoints = useMemo(() => {

    const pts: THREE.Vector3[] = []

    const minDistance = 0.008


    scene.traverse((child: any) => {

      if (child.isMesh) {

        const pos = child.geometry.attributes.position


        for (let i = 0; i < pos.count; i++) {

          const v = new THREE.Vector3(

            pos.getX(i),

            pos.getY(i),

            pos.getZ(i)

          )


          let tooClose = false

          for (let j = 0; j < pts.length; j++) {

            if (v.distanceTo(pts[j]) < minDistance) {

              tooClose = true

              break

            }

          }


          if (!tooClose) pts.push(v)

        }

      }

    })


    pts.sort(() => Math.random() - 0.5)

    return pts

  }, [scene])


  // --------------------------------------------------

  // CONTRIBUTION-BASED ORDER

  // --------------------------------------------------


  const points = useMemo(() => {

    if (rawPoints.length === 0) return []


    const used = new Set<number>()

    const ordered: THREE.Vector3[] = []


    const getRandomUnused = () => {

      let idx = 0

      do {

        idx = Math.floor(Math.random() * rawPoints.length)

      } while (used.has(idx))

      used.add(idx)

      return rawPoints[idx]

    }


    for (const c of contributions) {

      // SINGLE → random

      if (c.quantity === 1) {

        ordered.push(getRandomUnused())

        continue

      }


      // MULTI → cluster

      const anchorIndex = Math.floor(Math.random() * rawPoints.length)

      const anchor = rawPoints[anchorIndex]


      const nearby = rawPoints

        .map((p, i) => ({ p, i, d: p.distanceToSquared(anchor) }))

        .sort((a, b) => a.d - b.d)


      let added = 0


      for (let i = 0; i < nearby.length && added < c.quantity; i++) {

        const idx = nearby[i].i

        if (!used.has(idx)) {

          used.add(idx)

          ordered.push(nearby[i].p)

          added++

        }

      }

    }


    return ordered

  }, [rawPoints, contributions])


  // --------------------------------------------------


  useFrame((state, delta) => {

    if (!spinRef.current || !baseRef.current || !glowRef.current) return


    spinRef.current.rotation.z = state.clock.elapsedTime * 0.02


    if (lastPulseTriggerRef.current !== skullPulseTrigger) {

      lastPulseTriggerRef.current = skullPulseTrigger

      structurePulseRef.current = 1

    }


    structurePulseRef.current = Math.max(0, structurePulseRef.current - delta * 0.9)


    const targetCount = Math.min(points.length, 50000)


    if (displayCountRef.current < targetCount) {

      displayCountRef.current += 6

    }


    let glowCount = 0


    for (let i = 0; i < displayCountRef.current && i < targetCount; i++) {

      const p = points[i]


      const pulse = Math.sin(state.clock.elapsedTime * 0.8 + i) * 0.5 + 0.5

      const scale = 1 + pulse * 0.15


      tempObject.position.copy(p)

      tempObject.scale.setScalar(scale * 0.9)

      tempObject.updateMatrix()

      baseRef.current.setMatrixAt(i, tempObject.matrix)


      tempColor.copy(baseColor).lerp(shimmerColor, pulse)

      baseRef.current.setColorAt(i, tempColor)


      // glow

      const glowStrength = pulse * 0.5 + structurePulseRef.current * 1.5


      if (glowStrength > 0.1) {

        tempObject.scale.setScalar(scale * 1.4)

        tempObject.updateMatrix()

        glowRef.current.setMatrixAt(glowCount, tempObject.matrix)


        tempColor.copy(shimmerColor)

        glowRef.current.setColorAt(glowCount, tempColor)


        glowCount++

      }

    }


    baseRef.current.count = displayCountRef.current

    glowRef.current.count = glowCount


    baseRef.current.instanceMatrix.needsUpdate = true

    glowRef.current.instanceMatrix.needsUpdate = true


    if (baseRef.current.instanceColor) {

      baseRef.current.instanceColor.needsUpdate = true

    }


    if (glowRef.current.instanceColor) {

      glowRef.current.instanceColor.needsUpdate = true

    }

  })


  return (

    <group rotation={[Math.PI / 2, 0, 0]}>

      <group ref={spinRef}>

        <instancedMesh ref={baseRef} args={[undefined, undefined, 50000]}>

          <sphereGeometry args={[0.003, 20, 20]} />

          <meshPhysicalMaterial

            ref={baseMaterialRef}

            emissive="#3567f1"

            emissiveIntensity={0.7}

            roughness={0.4}

            metalness={0.7}

            vertexColors

          />

        </instancedMesh>


        <instancedMesh ref={glowRef} args={[undefined, undefined, 50000]}>

          <sphereGeometry args={[0.003, 20, 20]} />

          <meshBasicMaterial

            ref={glowMaterialRef}

            transparent

            opacity={0.5}

            blending={THREE.AdditiveBlending}

            vertexColors

          />

        </instancedMesh>

      </group>

    </group>

  )

}
